```tsx 
"use client";
import React from "react";
import { motion } from "framer-motion";

const SpinDemo = () => {
  const circles = [
    { size: 100, duration: 3 },
    { size: 80, duration: 2.5 },
    { size: 60, duration: 2 },
    { size: 40, duration: 1.5 },
    { size: 20, duration: 1 },
    { size: 10, duration: 0.5 },
  ];

  return (
    <div className="relative h-64 w-64 flex items-center justify-center">
      {circles.map((circle, index) => (
        <motion.div
          key={index}
          style={{
            width: circle.size,
            height: circle.size,
            border: "3px dashed",
            borderRadius: "50%",
            borderTopWidth: "4px ",
            borderBottomWidth: "2px",
            borderColor: "var(--ring)",
            borderRightColor: "var(--muted-foreground)",
            position: "absolute",
          }}
          initial={{ rotate: -360 }}
          animate={{ rotate: 360 }}
          transition={{
            duration: circle.duration,
            repeat: Infinity,
            repeatType: "reverse",
            ease: "easeInOut",
            type: "tween",
          }}
        />
      ))}
    </div>
  );
};

export default SpinDemo;

```
